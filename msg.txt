# Environment variables Error class
class EnvironmentVariableError(Exception):
    """
    Hola
    """

# check_vars_environment
def check_vars_environment(IP_ADDRESS, PORT) -> bool:
    """
    Check Vars Environment from config.env
    Default Vars:
        IP_ADDRESS=0.0.0.0
        PORT=5634
    Args:
        IP_ADDRESS:
        PORT:
    Returns:
        bool: 
    """
    try:
        ip_address = os.getenv(IP_ADDRESS)
        port = os.getenv(PORT)
        if ip_address is None or port is None:
            return False
    except EnvironmentVariableError as error:
        logging.warning(f'Error at check_vars_environment {error}')
    finally:
        pass
